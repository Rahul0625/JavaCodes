SUM OF DIGITS

import java.util.*;
public class Main
{
	public static void main(String[] args) {
		System.out.println("sum of digits");
		Scanner in=new Scanner(System.in);
		int a=in.nextInt();
		int sum=0;
		while(a!=0){
		    sum+=a%10;
		    a=a/10;
		}
		System.out.print(sum);
	}
}
