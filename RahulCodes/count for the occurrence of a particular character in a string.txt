public class Main {

    public static void main(String[] args) {
        String s = "yysdfysdyfsdfdsyfydfysfysf";
        char c='y';
        int count=0;
        for(int i=0;i<s.length();i++){
            char ch=s.charAt(i);
            if(ch==c){
                count++;
            }
        }
        System.out.println(count);
    }
}
