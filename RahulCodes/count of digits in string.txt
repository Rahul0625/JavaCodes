public class Main {

    public static void main(String[] args) {
        String s = "Hello1234569854785966world";
        int noOfNum=0;
        for(int i=0;i<s.length();i++){
            char ch=s.charAt(i);
            if(Character.isDigit(ch)){
                noOfNum++;
            }
        }
        System.out.println(noOfNum);
    }
}
