public class Main {

    public static void main(String[] args) {
        String s = "sdxadd6asd5asdas2das3dsa6dad46da8das3d";
        int sum=0;
        for(int i=0;i<s.length();i++){
            char c=s.charAt(i);
            if(Character.isDigit(c)){
                sum+=Character.getNumericValue(c);
            }
        }
        System.out.println(sum);
    }
}
