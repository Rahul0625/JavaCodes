public class Main {

    public static void main(String[] args) {
        String s = "sdlkfsdlkfsaccoijwsqwqwsreauioscnd";
        int countVowels=0;
        char[] ch=s.toCharArray();
        for(int i=0;i<ch.length;i++){
            char c=ch[i];
            if(c=='a'||c=='e'||c=='i'||c=='o'||c=='u' || c=='A'||c=='E'|| c=='I'||c=='O'||c=='U'){
                countVowels++;
            }
        }
        System.out.println(countVowels);
    }
}
