public class Main {

    public static void main(String[] args) {
        String s = "MADAM";
        String e="";
        for(int i =s.length()-1;i>=0;i--){
            e=e+s.charAt(i);
        }
        if(e.equals(s)){
            System.out.println("THE GIVEN STRING IS PALINDROME");
        }
        else{
            System.out.println("NOT A PALINDROME");
        }
    }
}
