package com.company;
import java.util.*;
public class Main {
    public static void main(String[] args) {
        Scanner in =new Scanner(System.in);
        int a=in.nextInt();
        int b=in.nextInt();
        int max=(a>b)? a:b;
        int lcm=0;
        for(int i=max;i<=a*b;i++){
            if(i%a==0 &&i%b==0){
                lcm=i;
                break;
            }
        }
        System.out.println(lcm);
    }
}
